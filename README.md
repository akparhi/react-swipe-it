## Install

```bash
npm install --save react-swipe-it
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-swipe-it'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
